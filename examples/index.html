gulp-open example

var gulp = require("gulp");
var open = require("gulp-open");

// Open an URL

gulp.task("open", function(){
  var options = {
    url: "http://google.com",
    app: "google-chrome"
  };
  gulp.src("./")
  .pipe(open(options));
});

// Open a File:

gulp.task("open", function(){
  var options = {
    url: "./index.html",
    app: "firefox"
  };
  gulp.src("./")
  .pipe(open(options));
});



// Run the gulp task:

gulp.task("default", function(){
  gulp.run("open");
});
